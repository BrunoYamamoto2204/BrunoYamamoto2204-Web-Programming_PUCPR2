    `ID_Depto` INT PRIMARY KEY , 
    `Nome_Depto` VARCHAR(50) NOT NULL, 
    `Local` VARCHAR(50)

CREATE TABLE `Funcionario` (
    Matricula   VARCHAR(10) PRIMARY KEY,  
    Nome        VARCHAR(50),         
    CPF         VARCHAR(15),         
    DataNasc    DATE,                
    Email       VARCHAR(50),         
    CEP         VARCHAR(10),
    fk_ID_Depto INT, 
    FOREIGN KEY (`fk_ID_Depto`) REFERENCES `Departamento`(`ID_Depto`) /* FK -> PK */
);
